Actor Block_Weapon : Weapon
{
  Weapon.ammouse 1
  Weapon.AmmoGive 1
}

Actor Block_Ammo : Ammo
{
  Inventory.DefMaxAmount 999999
  Inventory.MaxAmount 999999
}

Actor Block_Bedrock_Ammo : Block_Ammo
{
  Tag "Bedrock"
  Inventory.Icon BLCKA0
  Inventory.PickupMessage "Bedrock"
}

actor Block_Bedrock_Puff
{
  +NoGravity
  Radius 1
  Height 2
  -Solid
  States
  {
    Spawn:
      TNT1 A 1 ACS_NamedExecute("place_block", 0)
      Stop
  }
}

actor Block_Bedrock_Weapon : Block_Weapon
{
  Tag "Bedrock"
  Inventory.Icon BLCKA0
  Inventory.PickupMessage "Bedrock"

	Weapon.ammotype "Block_Bedrock_Ammo"
    
  States
  {
    Spawn:
      BLCK A -1
      Stop
		Ready:
			BLCK A 1 A_WeaponReady
			Loop
		Select:
			BLCK A 1 A_Raise
			Loop
		Deselect:
			BLCK A 1 A_Lower
			Loop
		Fire:
			BLCK A 8 A_CustomPunch(0, TRUE, 0, "Block_Bedrock_Puff", 96)
			Goto Ready
  }
}
